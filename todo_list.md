    def classify_stage(self, item: OSBWatchlistItem, ohlcv: list) -> int:
        """마크 미너비니 기준 주가 단계 판정 (1~4단계)."""
        current = item.prev_close
        ma50, ma150, ma200 = item.ma_50d, item.ma_150d, item.ma_200d
        w52_high, w52_low = item.w52_hgpr, item.w52_lwpr

        # 🚨 4단계: 하락 국면 (가장 먼저 체크하여 배제)
        if current < ma200 or self._get_ma_slope(ohlcv, 200) < 0:
            return 4

        # 🚀 2단계: 상승 국면 (미너비니 트렌드 템플릿)
        is_trend_template_ok = (
            current > ma150 > ma200 and
            ma50 > ma150 and
            self._get_ma_slope(ohlcv, 200, days=20) > 0 and  # 200일선 우상향
            current > w52_low * 1.25 and                    # 바닥 대비 25% 상승
            current > w52_high * 0.75                       # 고점 대비 -25% 이내
        )
        if is_trend_template_ok:
            return 2

        # 🎢 3단계: 고점 국면
        # (2단계 조건을 일부 상실하고 변동성이 커지며 ma50을 건드리는 시점)
        if current < ma50 and self._get_volatility(ohlcv) > threshold:
            return 3

        # 💤 1단계: 무관심/바닥 다지기
        return 1

      4. 코드 개선 아이디어 (Python 로직)현재 코드의 _compute_rs_scores 메서드에 가중 수익률 개념을 도입하여 오닐의 방식에 더 가깝게 개선할 수 있습니다.$$RS = (C_0 x 2) + C_1 + C_2 + C_3$$(여기서 $C_0$는 최근 분기 수익률, $C_1, C_2, C_3$는 이전 분기 수익률입니다.)Python# 가중 RS 계산 예시
      def calc_weighted_rs(self, ohlcv):
          # 최근 1년(252일) 데이터를 4개 분기로 나눔
          q1 = self._calc_return(ohlcv[-63:])   # 최근 3개월 (가중치 40%)
          q2 = self._calc_return(ohlcv[-126:-63])
          q3 = self._calc_return(ohlcv[-189:-126])
          q4 = self._calc_return(ohlcv[-252:-189])
          
          # 최근 성과에 더 높은 가중치를 부여
          weighted_return = (q1 * 0.4) + (q2 * 0.2) + (q3 * 0.2) + (q4 * 0.2)
          return weighted_return
